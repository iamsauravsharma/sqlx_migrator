174enum PlanType {
175    Apply,
176    Revert,

179/// Struct that determines the type of migration plan to execute.
180///
181/// A [`Plan`] can specify whether to apply or revert migrations, and may target
182/// all migrations, specific migrations, or a limited number of migrations.
183#[derive(Debug)]
184pub struct Plan {
185    plan_type: PlanType,
186    app_migration: Option<(String, Option<String>)>,
187    count: Option<usize>,
188    fake: bool,
189}
190
191impl Plan {
192    fn new(
193        plan_type: PlanType,
194        app_migration: Option<(String, Option<String>)>,
195        count: Option<usize>,
196    ) -> Self {
197        Self {
198            plan_type,
199            app_migration,
200            count,
201            fake: false,
202        }
203    }
204
205    /// Sets the plan as a "fake" plan.
206    ///
207    /// When the plan is marked as fake, the migration status is updated to
208    /// either "applied" or "reverted" without actually performing any
209    /// migration operations. This is useful for scenarios where you want to
210    /// simulate the effect of applying or reverting a migration, but
211    /// without making changes to the database.
212    ///
213    /// By default, the `fake` flag is set to `false`, and the migration
214    /// operations are executed as expected.
215    #[must_use]
216    pub fn fake(self, fake: bool) -> Self {
217        let mut plan = self;
218        plan.fake = fake;
219        plan
220    }
221
222    /// Creates a new plan to apply all migrations.
223    #[must_use]
224    pub fn apply_all() -> Self {
225        Self::new(PlanType::Apply, None, None)
226    }
227
228    /// Creates a new plan to apply a specific migration by name. If migration
229    /// name is not provided it will apply app all migrations
230    #[must_use]
231    pub fn apply_name(app: &str, name: &Option<String>) -> Self {
232        Self::new(PlanType::Apply, Some((app.to_string(), name.clone())), None)
233    }
234
235    /// Creates a new plan to apply a limited number of migrations.
236    #[must_use]
237    pub fn apply_count(count: usize) -> Self {
238        Self::new(PlanType::Apply, None, Some(count))
239    }
240
241    /// Creates a new plan to revert all migrations.
242    #[must_use]
243    pub fn revert_all() -> Self {
244        Self::new(PlanType::Revert, None, None)
245    }
246
247    /// Creates a new plan to revert a specific migration by name. If migration
248    /// name is not provided it will revert app all migrations
249    #[must_use]
250    pub fn revert_name(app: &str, name: &Option<String>) -> Self {
251        Self::new(
252            PlanType::Revert,
253            Some((app.to_string(), name.clone())),
254            None,
255        )
256    }
257
258    /// Creates a new plan to revert a limited number of migrations.
259    #[must_use]
260    pub fn revert_count(count: usize) -> Self {
261        Self::new(PlanType::Revert, None, Some(count))
262    }
263}

265/// The [`Info`] trait provides database-agnostic methods for managing
266/// migrations and interacting with migration states.
267pub trait Info<DB> {
268    /// Returns a reference to the list of migrations.
269    fn migrations(&self) -> &Vec<BoxMigration<DB>>;
270
271    /// Returns a mutable reference to the list of migrations.
272    fn migrations_mut(&mut self) -> &mut Vec<BoxMigration<DB>>;
273
274    /// Adds a list of migrations to the migrator.
275    ///
276    /// This method accepts a vector of migrations and adds each one
277    /// individually to ensure proper handling of migration relationships
278    /// and duplicates.
279    fn add_migrations(&mut self, migrations: Vec<BoxMigration<DB>>) {
280        for migration in migrations {
281            self.add_migration(migration);
282        }
283    }
284
285    /// Adds a single migration to the migrator.
286    fn add_migration(&mut self, migration: BoxMigration<DB>) {
287        // if virtual migration is present in list with same app and name than remove
288        // virtual migration from list
289        if let Some(migration_index) = self
290            .migrations()
291            .iter()
292            .position(|elem| elem == &migration && elem.is_virtual() && !migration.is_virtual())
293        {
294            self.migrations_mut().remove(migration_index);
295        }
296
297        // check if migration is already added or not we want to use vec here even if
298        // hash set can be used but hash set do not have consistent order which may
299        // bring issue such as plan may be different between between dry run and
300        // actually running migration
301        if !self.migrations().contains(&migration) {
302            // ignore parents, replaces and run before for virtual migration
303            if migration.is_virtual() {
304                self.migrations_mut().push(migration);
305            } else {
306                let migration_parents = migration.parents();
307                let migration_replaces = migration.replaces();
308                let migration_run_before = migration.run_before();
309
310                self.migrations_mut().push(migration);
311
312                for parent in migration_parents {
313                    self.add_migration(parent);
314                }
315                for replace in migration_replaces {
316                    self.add_migration(replace);
317                }
318                for run_before in migration_run_before {
319                    self.add_migration(run_before);
320                }
321            }
322        }
323    }
324}

326/// The [`DatabaseOperation`] trait defines a set of methods for performing
327/// operations related to migration management on the database.
328///
329/// This trait is typically implemented for a database to support migration
330/// operations, such as ensuring the migration table exists, adding or
331/// removing migrations from the table, and locking the database during
332/// migration processes.
333#[async_trait::async_trait]
334pub trait DatabaseOperation<DB>
335where
336    DB: Database,
337{
338    /// Ensure migration table is created before running migrations. If not
339    /// create one
340    async fn ensure_migration_table_exists(
341        &self,
342        connection: &mut <DB as Database>::Connection,
343    ) -> Result<(), Error>;
344
345    /// Drop migration table if migration table exists
346    async fn drop_migration_table_if_exists(
347        &self,
348        connection: &mut <DB as Database>::Connection,
349    ) -> Result<(), Error>;
350
351    /// Adds a migration record to the migration table in the database.
352    async fn add_migration_to_db_table(
353        &self,
354        connection: &mut <DB as Database>::Connection,
355        migration: &BoxMigration<DB>,
356    ) -> Result<(), Error>;
357
358    /// Removes a migration record from the migration table in the database.
359    async fn delete_migration_from_db_table(
360        &self,
361        connection: &mut <DB as Database>::Connection,
362        migration: &BoxMigration<DB>,
363    ) -> Result<(), Error>;
364
365    /// Fetches the list of applied migrations from the migration table in the
366    /// database.
367    async fn fetch_applied_migration_from_db(
368        &self,
369        connection: &mut <DB as Database>::Connection,
370    ) -> Result<Vec<AppliedMigrationSqlRow>, Error>;
371
372    /// Lock database while doing migrations so no two migrations run together
373    async fn lock(&self, connection: &mut <DB as Database>::Connection) -> Result<(), Error>;
374
375    /// Unlock locked database
376    async fn unlock(&self, connection: &mut <DB as Database>::Connection) -> Result<(), Error>;
377}

379fn populate_replace_recursive<'populate, DB>(
380    replace_hash_map: &mut HashMap<&'populate BoxMigration<DB>, Vec<&'populate BoxMigration<DB>>>,
381    key: &'populate BoxMigration<DB>,
382    value: &'populate BoxMigration<DB>,
383) -> Result<(), Error> {
384    // protect against a case where two migration replaces each other
385    if key == value {
386        return Err(Error::PlanError {
387            message: "two migrations replaces each other".to_string(),
388        });
389    }
390    let replace_hash_map_vec = replace_hash_map.entry(key).or_default();
391    if !replace_hash_map_vec.contains(&value) {
392        replace_hash_map_vec.push(value);
393    }
394    if let Some(grand_values) = replace_hash_map.clone().get(value) {
395        for grand_value in grand_values {
396            populate_replace_recursive(replace_hash_map, key, grand_value)?;
397        }
398    }
399    Ok(())
400}

402fn get_parent_recursive<DB>(
403    migration: &BoxMigration<DB>,
404    original_migration: &[BoxMigration<DB>],
405) -> Result<Vec<BoxMigration<DB>>, Error> {
406    let mut parents: Vec<BoxMigration<DB>> = vec![];
407    for parent in migration.parents() {
408        parents.push(Box::new((
409            parent.app().to_string(),
410            parent.name().to_string(),
411        )));
412        let found_parent = if parent.is_virtual() {
413            original_migration
414                .iter()
415                .find(|&search_parent| search_parent == &parent)
416                .ok_or(Error::PlanError {
417                    message: "failed to find parent non virtual migration".to_string(),
418                })?
419        } else {
420            &parent
421        };
422        parents.extend(get_parent_recursive(found_parent, original_migration)?);
423    }
424    Ok(parents)
425}

427fn get_run_before_recursive<DB>(
428    migration: &BoxMigration<DB>,
429    original_migration: &[BoxMigration<DB>],
430) -> Result<Vec<BoxMigration<DB>>, Error> {
431    let mut run_before_list: Vec<BoxMigration<DB>> = vec![];
432    for run_before in migration.run_before() {
433        run_before_list.push(Box::new((
434            run_before.app().to_string(),
435            run_before.name().to_string(),
436        )));
437        let found_run_before = if run_before.is_virtual() {
438            original_migration
439                .iter()
440                .find(|&search_run_before| search_run_before == &run_before)
441                .ok_or(Error::PlanError {
442                    message: "failed to find run before non virtual migration".to_string(),
443                })?
444        } else {
445            &run_before
446        };
447        run_before_list.extend(get_parent_recursive(found_run_before, original_migration)?);
448    }
449    Ok(run_before_list)
450}

452// filter migration list to only contains migrations which is related to with
453// list migration, removes all migrations which is not related to them according
454// to provided plan. We should not check replaces migration since it is already
455// handled and all replaces migration are removed as required
456fn only_related_migration<DB>(
457    migration_list: &mut MigrationVec<DB>,
458    with_list: Vec<&BoxMigration<DB>>,
459    plan_type: &PlanType,
460    original_migration: &[BoxMigration<DB>],
461) -> Result<(), Error> {
462    let mut related_migrations = vec![];
463    for with in with_list {
464        // check if with migrations is already added or not. Sometimes with list
465        // contains migrations which are interrelated so we do not need to add
466        // already added migration again
467        if !related_migrations.contains(&with) {
468            related_migrations.push(with);
469            match plan_type {
470                PlanType::Apply => {
471                    let with_parents = get_parent_recursive(with, original_migration)?;
472                    for &migration in migration_list.iter() {
473                        if !related_migrations.contains(&migration)
474                            && (with_parents.contains(migration)
475                                || get_run_before_recursive(migration, original_migration)?
476                                    .contains(with))
477                        {
478                            related_migrations.push(migration);
479                        }
480                    }
481                }
482                PlanType::Revert => {
483                    let with_run_before = get_run_before_recursive(with, original_migration)?;
484                    for &migration in migration_list.iter() {
485                        if !related_migrations.contains(&migration)
486                            && (with_run_before.contains(migration)
487                                || get_parent_recursive(migration, original_migration)?
488                                    .contains(with))
489                        {
490                            related_migrations.push(migration);
491                        }
492                    }
493                }
494            }
495        }
496    }
497    migration_list.retain(|&migration| related_migrations.contains(&migration));
498    Ok(())
499}

501/// Process plan to provided migrations list
502fn process_plan<DB>(
503    migration_list: &mut MigrationVec<DB>,
504    applied_migrations: &MigrationVec<DB>,
505    plan: &Plan,
506    original_migration: &[BoxMigration<DB>],
507) -> Result<(), Error>
508where
509    DB: Database,
510{
511    // Modify migration list according to plan type
512    match plan.plan_type {
513        PlanType::Apply => {
514            migration_list.retain(|migration| !applied_migrations.contains(migration));
515        }
516        PlanType::Revert => {
517            migration_list.retain(|migration| applied_migrations.contains(migration));
518            migration_list.reverse();
519        }
520    }
521
522    if let Some((app, migration_name)) = &plan.app_migration {
523        // Find position of last migration which matches condition of provided app and
524        // migration name
525        let position = if let Some(name) = migration_name {
526            let Some(pos) = migration_list
527                .iter()
528                .rposition(|migration| migration.app() == app && migration.name() == name)
529            else {
530                if migration_list
531                    .iter()
532                    .any(|migration| migration.app() == app)
533                {
534                    return Err(Error::PlanError {
535                        message: format!("migration {app}:{name} doesn't exists for app"),
536                    });
537                }
538                return Err(Error::PlanError {
539                    message: format!("app {app} doesn't exists"),
540                });
541            };
542            pos
543        } else {
544            let Some(pos) = migration_list
545                .iter()
546                .rposition(|migration| migration.app() == app)
547            else {
548                return Err(Error::PlanError {
549                    message: format!("app {app} doesn't exists"),
550                });
551            };
552            pos
553        };
554        migration_list.truncate(position + 1);
555        let with_list = if migration_name.is_some() {
556            vec![migration_list[position]]
557        } else {
558            migration_list
559                .iter()
560                .filter(|pos_migration| pos_migration.app() == app)
561                .copied()
562                .collect::<Vec<_>>()
563        };
564        only_related_migration(
565            migration_list,
566            with_list,
567            &plan.plan_type,
568            original_migration,
569        )?;
570    } else if let Some(count) = plan.count {
571        let actual_len = migration_list.len();
572        if count > actual_len {
573            return Err(Error::PlanError {
574                message: format!(
575                    "passed count value is larger than migration length: {actual_len}"
576                ),
577            });
578        }
579        migration_list.truncate(count);
580    }
581    Ok(())
582}

584fn get_recursive<'get, DB>(
585    hash_map: &'get HashMap<BoxMigration<DB>, &'get BoxMigration<DB>>,
586    val: &'get BoxMigration<DB>,
587) -> Vec<&'get BoxMigration<DB>> {
588    let mut recursive_vec = vec![val];
589    if let Some(&parent) = hash_map.get(val) {
590        recursive_vec.extend(get_recursive(hash_map, parent));
591    }
592    recursive_vec
593}

595/// The [`Migrate`] trait defines methods to manage and apply database
596/// migrations according to a given plan.
597///
598/// This trait combines the functionalities of the [`Info`] and
599/// [`DatabaseOperation`] traits, providing a full set of migration
600/// capabilities. All methods have default implementations, meaning no explicit
601/// implementation is required. Additionally, all methods are database-agnostic.
602#[async_trait::async_trait]
603pub trait Migrate<DB>: Info<DB> + DatabaseOperation<DB> + Send + Sync
604where
605    DB: Database,
606{
607    /// Generate migration plan according to plan.
608    ///
609    /// Returns a vector of migration. If plan is none than it will generate
610    /// plan with all migrations in order of apply
611    #[expect(clippy::too_many_lines)]
612    async fn generate_migration_plan(
613        &self,
614        connection: &mut <DB as Database>::Connection,
615        plan: Option<&Plan>,
616    ) -> MigrationVecResult<DB> {
617        if self.migrations().is_empty() {
618            return Err(Error::PlanError {
619                message: "no migration are added to migration list".to_string(),
620            });
621        }
622        if self
623            .migrations()
624            .iter()
625            .any(|migration| migration.is_virtual())
626        {
627            return Err(Error::PlanError {
628                message: "virtual migrations which is not replaced is present".to_string(),
629            });
630        }
631
632        tracing::debug!("generating {:?} migration plan", plan);
633
634        // Hashmap which contains key as migration and value is migration which replaces
635        // this migration. One migration can only have one parent
636        let mut replaces_child_parent_hash_map = HashMap::new();
637
638        for parent_migration in self.migrations() {
639            for child_migration in parent_migration.replaces() {
640                let child_name = format!("{}:{}", child_migration.app(), child_migration.name());
641                if replaces_child_parent_hash_map
642                    .insert(child_migration, parent_migration)
643                    .is_some()
644                {
645                    return Err(Error::PlanError {
646                        message: format!("migration {child_name} replaced multiple times",),
647                    });
648                }
649            }
650        }
651
652        // Hashmap which contains all children of migration generated from replace list
653        let mut replace_children = HashMap::<_, Vec<_>>::new();
654        // in first loop add initial parent and child from parent due to replace
655        for (child, &parent) in &replaces_child_parent_hash_map {
656            let children_migration = if child.is_virtual() {
657                self.migrations()
658                    .iter()
659                    .find(|&search_migration| search_migration == child)
660                    .ok_or(Error::PlanError {
661                        message: "Failed finding non virtual migration for virtual migration"
662                            .to_string(),
663                    })?
664            } else {
665                child
666            };
667            replace_children
668                .entry(parent)
669                .or_default()
670                .push(children_migration);
671        }
672        // in second loop through recursive add all descendants
673        for (child, &parent) in &replaces_child_parent_hash_map {
674            let children_migration = if child.is_virtual() {
675                self.migrations()
676                    .iter()
677                    .find(|&search_migration| search_migration == child)
678                    .ok_or(Error::PlanError {
679                        message: "Failed finding non virtual migration for virtual migration"
680                            .to_string(),
681                    })?
682            } else {
683                child
684            };
685            populate_replace_recursive(&mut replace_children, parent, children_migration)?;
686        }
687        // Hashmap which contains key as migration and value as list of migration
688        // which becomes parent for key due to value having key as run before value
689        let mut run_before_child_parent_hash_map = HashMap::<_, Vec<_>>::new();
690
691        for parent_migration in self.migrations() {
692            for run_before_migration in parent_migration.run_before() {
693                run_before_child_parent_hash_map
694                    .entry(run_before_migration)
695                    .or_default()
696                    .push(parent_migration);
697            }
698        }
699
700        let mut migration_list = Vec::new();
701
702        // Create migration list until migration list length is equal to original vec
703        // length
704        let original_migration_length = self.migrations().len();
705        while migration_list.len() != original_migration_length {
706            let loop_initial_migration_list_length = migration_list.len();
707            for migration in self.migrations() {
708                let all_required_added = !migration_list.contains(&migration)
709                    && migration
710                        .parents()
711                        .iter()
712                        .all(|parent_migration| migration_list.contains(&parent_migration))
713                    && run_before_child_parent_hash_map
714                        .get(migration)
715                        .unwrap_or(&vec![])
716                        .iter()
717                        .all(|run_before_migration| migration_list.contains(run_before_migration))
718                    && replaces_child_parent_hash_map
719                        .get(migration)
720                        .is_none_or(|replace_migration| migration_list.contains(replace_migration))
721                    && replace_children.get(migration).is_none_or(|children| {
722                        // check if children parents and run before are added or not already before
723                        // adding replace migration. Since replace migration may not depend on
724                        // children parent its need to be added first
725                        children.iter().all(|&child| {
726                            child
727                                .parents()
728                                .iter()
729                                .all(|child_parent| migration_list.contains(&child_parent))
730                                && run_before_child_parent_hash_map
731                                    .get(child)
732                                    .unwrap_or(&vec![])
733                                    .iter()
734                                    .all(|run_before_migration| {
735                                        migration_list.contains(run_before_migration)
736                                            || children.contains(run_before_migration)
737                                    })
738                        })
739                    });
740                if all_required_added {
741                    migration_list.push(migration);
742                }
743            }
744
745            // If old migration plan length is equal to current length than no new migration
746            // was added. Next loop also will not add migration so return error. This case
747            // can arise due to looping in migration plan i.e If there is two migration A
748            // and B, than when B is ancestor of A as well as descendants of A
749            if loop_initial_migration_list_length == migration_list.len() {
750                return Err(Error::PlanError {
751                    message: "reached deadlock stage during plan generation".to_string(),
752                });
753            }
754        }
755
756        // if there is only plan than further process. In further process replaces
757        // migrations are also handled for removing conflicting migrations where certain
758        // migrations replaces certain other migrations. While initially creating
759        // migrations both new and replaced migration are present
760        if let Some(some_plan) = plan {
761            self.ensure_migration_table_exists(connection).await?;
762
763            let applied_migration_sql_rows =
764                self.fetch_applied_migration_from_db(connection).await?;
765
766            // convert applied migration sql rows to vector of migration implemented
767            // objects
768            let mut applied_migrations = Vec::new();
769            for migration in self.migrations() {
770                if applied_migration_sql_rows
771                    .iter()
772                    .any(|sqlx_migration| sqlx_migration == migration)
773                {
774                    applied_migrations.push(migration);
775                }
776            }
777
778            // Check if any of parents of certain applied migrations are applied or not. If
779            // any parents are not applied for applied migration than raises
780            // error also takes consideration of replace migration
781            for &migration in &applied_migrations {
782                let mut parents = vec![];
783                if let Some(run_before_list) = run_before_child_parent_hash_map.get(migration) {
784                    for &run_before in run_before_list {
785                        parents.push(run_before);
786                    }
787                }
788                let main_parents = migration.parents();
789                for parent in &main_parents {
790                    parents.push(parent);
791                }
792                for parent in parents {
793                    let recursive_vec = get_recursive(&replaces_child_parent_hash_map, parent);
794                    if !applied_migrations
795                        .iter()
796                        .any(|applied| recursive_vec.contains(applied))
797                    {
798                        return Err(Error::PlanError {
799                            message: format!(
800                                "children migration {}:{} applied before its parent migration \
801                                 {}:{}",
802                                migration.app(),
803                                migration.name(),
804                                parent.app(),
805                                parent.name()
806                            ),
807                        });
808                    }
809                }
810            }
811
812            // Remove migration from migration list according to replaces vector
813            for migration in migration_list.clone() {
814                // Only need to check case when migration have children
815                if let Some(children) = replace_children.get(&migration) {
816                    // Check if any replaces children are applied or not
817                    let replaces_applied = children
818                        .iter()
819                        .any(|&replace_migration| applied_migrations.contains(&replace_migration));
820
821                    // If any one of replaced migrations is applied than do not add current
822                    // migration to migration plan else add only current migration to migration plan
823                    if replaces_applied {
824                        // Error if current migration as well as replace migration both are applied
825                        if applied_migrations.contains(&migration) {
826                            return Err(Error::PlanError {
827                                message: format!(
828                                    "migration {}:{} and its replaces are applied together",
829                                    migration.app(),
830                                    migration.name(),
831                                ),
832                            });
833                        }
834                        migration_list.retain(|&plan_migration| migration != plan_migration);
835                    } else {
836                        // we can remove all children migrations here since migrations which
837                        // replaced them will be above them in generation list so migration will
838                        // apply in provided order
839                        for replaced_migration in children {
840                            migration_list
841                                .retain(|plan_migration| replaced_migration != plan_migration);
842                        }
843                    }
844                }
845            }
846
847            process_plan(
848                &mut migration_list,
849                &applied_migrations,
850                some_plan,
851                self.migrations(),
852            )?;
853        }
854
855        Ok(migration_list)
856    }
857
858    /// Run provided plan migrations
859    ///
860    /// # Errors
861    /// If failed to run provided plan migrations
862    async fn run(
863        &self,
864        connection: &mut <DB as Database>::Connection,
865        plan: &Plan,
866    ) -> Result<(), Error> {
867        tracing::debug!("running plan {:?}", plan);
868        self.lock(connection).await?;
869        for migration in self.generate_migration_plan(connection, Some(plan)).await? {
870            match plan.plan_type {
871                PlanType::Apply => {
872                    tracing::debug!("applying {} : {}", migration.app(), migration.name());
873                    let operations = migration.operations();
874                    if migration.is_atomic() {
875                        let mut transaction = connection.begin().await?;
876                        if !plan.fake {
877                            for operation in operations {
878                                operation.up(&mut transaction).await?;
879                            }
880                        }
881                        self.add_migration_to_db_table(&mut transaction, migration)
882                            .await?;
883                        transaction.commit().await?;
884                    } else {
885                        if !plan.fake {
886                            for operation in operations {
887                                operation.up(connection).await?;
888                            }
889                        }
890                        self.add_migration_to_db_table(connection, migration)
891                            .await?;
892                    }
893                }
894                PlanType::Revert => {
895                    tracing::debug!("reverting {} : {}", migration.app(), migration.name());
896
897                    // Reverse operation since last applied operation need to be reverted first
898                    let mut operations = migration.operations();
899                    operations.reverse();
900
901                    if migration.is_atomic() {
902                        let mut transaction = connection.begin().await?;
903                        if !plan.fake {
904                            for operation in operations {
905                                operation.down(&mut transaction).await?;
906                            }
907                        }
908                        self.delete_migration_from_db_table(&mut transaction, migration)
909                            .await?;
910                        transaction.commit().await?;
911                    } else {
912                        if !plan.fake {
913                            for operation in operations {
914                                operation.down(connection).await?;
915                            }
916                        }
917                        self.delete_migration_from_db_table(connection, migration)
918                            .await?;
919                    }
920                }
921            }
922        }
923        self.unlock(connection).await?;
924        Ok(())
925    }
926}

928const DEFAULT_TABLE_NAME: &str = "_sqlx_migrator_migrations";

930/// Migrator struct which store migrations graph and information related to
931/// different library supported migrations
932pub struct Migrator<DB> {
933    migrations: Vec<BoxMigration<DB>>,
934    table_name: String,
935}
936
937impl<DB> Migrator<DB> {
938    /// Create new migrator
939    #[must_use]
940    pub fn new() -> Self {
941        Self {
942            migrations: Vec::default(),
943            table_name: DEFAULT_TABLE_NAME.to_string(),
944        }
945    }
946
947    /// Use prefix for migrator table name only ascii alpha numeric and
948    /// underscore characters are supported for table name. prefix will set
949    /// table name as `_{prefix}{default_table_name}` where default table
950    /// name is `_sqlx_migrator_migrations`
951    ///
952    /// # Errors
953    /// When passed prefix is not ascii alpha numeric or underscore character
954    pub fn with_prefix(mut self, prefix: impl Into<String>) -> Result<Self, Error> {
955        let prefix_str = prefix.into();
956        if !prefix_str
957            .chars()
958            .all(|c| char::is_ascii_alphanumeric(&c) || c == '_')
959        {
960            return Err(Error::NonAsciiAlphaNumeric);
961        }
962        self.table_name = format!("_{prefix_str}{DEFAULT_TABLE_NAME}");
963        Ok(self)
964    }
965
966    /// Get name of table which is used for storing migrations related
967    /// information in database
968    #[must_use]
969    pub fn table_name(&self) -> &str {
970        &self.table_name
971    }
972}
973
974impl<DB> Default for Migrator<DB> {
975    fn default() -> Self {
976        Self::new()
977    }
978}
